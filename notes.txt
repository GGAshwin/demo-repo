git remote <name> <repoName> #used to connect to a remote repository

Reminder!!
Everytime MASTER is mentioned anywhere substitue it with MAIN

git checkout #used to switch branches

git checkout -b <name> #used to create new branches and switch to it

git branch #used to view all the branches

git branch -d #used to delete an already merged branch

git pull <remote> <branch> #used to pull from the remote branch if the branch is not already set yet

git commit -am "message" #used to add and commit at the same time ONLY WORKS FOR MODIFIED FILES

git reset #used to undo the STAGED files

git reset HEAD~1 #used to reset to the last but one commit

git merge --abort #used to stop or revert the merging process
